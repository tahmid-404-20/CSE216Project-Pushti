CREATE OR REPLACE FUNCTION NO_OF_DISTRICTS_DIVISION(div_id IN INTEGER) RETURN NUMBER IS 
	counter NUMBER;
BEGIN
	SELECT COUNT(id) INTO counter 
	FROM District 
	WHERE division_id = div_id;
	RETURN counter;
END;
/


CREATE OR REPLACE FUNCTION NO_OF_UPAZILLAS_DIVISION(div_id IN INTEGER) RETURN NUMBER IS 
	counter NUMBER;
BEGIN
	SELECT SUM(NO_OF_UPAZILLAS_DISTRICT(id)) INTO counter 
	FROM District 
	WHERE division_id = div_id;

/
