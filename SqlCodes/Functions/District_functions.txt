CREATE OR REPLACE FUNCTION NO_OF_UPAZILLAS_DISTRICT(dis_id IN INTEGER) RETURN NUMBER IS 
	counter NUMBER;
BEGIN
	SELECT COUNT(id) INTO counter 
	FROM Upazilla 
	WHERE district_id = dis_id;
	RETURN counter;
END;
/

	counter NUMBER;
